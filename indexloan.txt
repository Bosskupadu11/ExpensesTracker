    <script>
        let loans = [
            { name: "Car Loan (Axia)", amount: 537, duration: 108, remaining: 108 },
            { name: "Motor Loan (LC 135)", amount: 232, duration: 60, remaining: 60 },
            { name: "Phone Loan", amount: 205, duration: 36, remaining: 36 }
        ];

        function renderLoans() {
            const table = document.getElementById("loanTable");
            table.innerHTML = "";
            let totalLoan = 0;
            loans.forEach((loan, index) => {
                totalLoan += loan.amount;
                const row = `<tr>
                    <td>${loan.name}</td>
                    <td>RM${loan.amount}</td>
                    <td>${loan.duration} months</td>
                    <td>${loan.remaining}</td>
                </tr>`;
                table.innerHTML += row;
            });
            document.getElementById("totalLoan").innerText = totalLoan;
        }
        
        renderLoans();
    </script>
